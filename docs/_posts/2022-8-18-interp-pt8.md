## Logic
When the VM encounters an if statement, it goes to a gathering-mode that pushes lines into the line buffer.
A matching "}" then stops this gathering. If condition was true the lines from the buffer get executed.
Else if and else also clear and initialize the buffer and possibly execute code if appropriate.

```rust
pub fn exec_line(vm: &mut VM, parsed_line: &Line) {
    if vm.scope != 0 { 
        // Means gathering mode is on
        if vm.scope == 1 {
            match parsed_line.type_ {
                LineType::If(_) => {
                    vm.scope = 2;
                    vm.line_buffer.push(parsed_line.clone());
                    return;
                }
                LineType::ElseIf(_) |
                LineType::Else |
                LineType::EndScope => { }
                _ => {
                    vm.line_buffer.push(parsed_line.clone());
                    return;
                }
            }
        } else {
            match parsed_line.type_ {
                LineType::If(_)    => vm.scope += 1,
                LineType::EndScope => vm.scope -= 1,
                _ => { }
            }
            vm.line_buffer.push(parsed_line.clone());
            return;
        }
    }
```
Here's the **exec_line** cases.
```rust
    match &parsed_line.type_ {

        ...

        LineType::If(cond_expr) => {
            let cond_value = expr_to_value(vm, cond_expr);
            if let value::Value::Boolean(cond) = cond_value {
                vm.scope_state = ScopeState::If(cond);
            } else {
                err::throw_loc(
                    err::Err::TokenExpected { expected: "condition".to_owned() },
                    &cond_expr.loc
                );
                vm.scope_state = ScopeState::If(false);
            }
            vm.scope = 1; // Into gathering mode
        }
        LineType::ElseIf(cond_expr) => {
            vm.scope = 0; // Out of gathering mode
            let prev_line_buffer = std::mem::replace(&mut vm.line_buffer, vec![]);
            let cond_value = expr_to_value(vm, cond_expr);
            if let ScopeState::If(prev_cond) = vm.scope_state {
                if let value::Value::Boolean(cond) = cond_value {
                    vm.scope_state = ScopeState::If(!prev_cond && cond);
                } else {
                    err::throw_loc(
                        err::Err::TokenExpected { expected: "condition".to_owned() },
                        &cond_expr.loc
                    );
                    vm.scope_state = ScopeState::If(false)
                }
                if prev_cond {
                    for line in prev_line_buffer {
                        exec_line(vm, &line);
                    }
                }
            } else {
                err::throw_loc(
                    err::Err::TokenUnexpected { not_expected: "else if".to_owned() },
                    &parsed_line.loc,
                );
            }
            vm.scope = 1;
        }
        LineType::Else => {
            vm.scope = 0;
            let prev_line_buffer = std::mem::replace(&mut vm.line_buffer, vec![]);
            if let ScopeState::If(prev_cond) = vm.scope_state {
                vm.scope_state = ScopeState::Else(!prev_cond);
                if prev_cond {
                    for line in prev_line_buffer {
                        exec_line(vm, &line);
                    }
                }
            } else {
                err::throw_loc(
                    err::Err::TokenUnexpected { not_expected: "else".to_owned() },
                    &parsed_line.loc,
                );
            }
            vm.scope = 1;
        }
        LineType::EndScope => {
            vm.scope = 0;
            let prev_line_buffer = std::mem::replace(&mut vm.line_buffer, vec![]);
            if let ScopeState::If(prev_cond) | ScopeState::Else(prev_cond) = vm.scope_state {
                if prev_cond {
                    for line in prev_line_buffer {
                        exec_line(vm, &line);
                    }
                }
                vm.scope_state = ScopeState::Regular
            } else {
                err::throw_loc(
                    err::Err::TokenUnexpected { not_expected: "\"}\"".to_owned() },
                    &parsed_line.loc,
                );
            }
        }
    }
}
```
